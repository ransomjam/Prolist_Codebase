import { useEffect, useState } from 'react';
import { useNavigate } from 'react-router-dom';

export default function AuctionPostProcess({ auctions, setAuctions, userSession, incrementVendorSales }) {
  const [closedAuctions, setClosedAuctions] = useState([]);
  const navigate = useNavigate();

  useEffect(() => {
    const now = new Date().getTime();
    auctions.forEach((auction) => {
      const end = new Date(auction.endTime).getTime();

      if (end <= now && auction.status !== 'Closed') {
        // Close auction
        const updatedAuctions = auctions.map(a =>
          a.id === auction.id ? { ...a, status: 'Closed' } : a
        );
        setAuctions(updatedAuctions);
        setClosedAuctions((prev) => [...prev, auction]);

        // Notify winner & vendor (simulate)
        if (userSession?.username === auction.highestBidder) {
          alert(`Congrats! You won the auction for ${auction.product.title}. Proceed to checkout.`);
          navigate(`/checkout/${auction.id}`);
        } else if (userSession?.username === auction.vendor) {
          alert(`Auction ended for ${auction.product.title}. Awaiting winner payment.`);
        }
      }
    });
  }, [auctions, userSession, navigate, setAuctions]);

  // Admin fund release simulation (to be built separately)

  return null; // This is a background component
}
